## Usage in Code

```tsx
import { useGLTF } from '@react-three/drei'

// Load a model
const { scene } = useGLTF('/assets/models/sword.glb')

// Load multiple models
const helmet = useGLTF('/assets/models/damaged-helmet.glb')
const dragon = useGLTF('/assets/models/dragon.glb')

// Preload for performance
useGLTF.preload('/assets/models/sword.glb')
```

## Procedural Character System

```tsx
import { GskCharacter } from './components/GskCharacter'
import { useMoodStore } from './stores/mood-store'

// In your scene:
const mood = useMoodStore((s) => s.mood)
<GskCharacter mood={mood} clothing="suit" position={[0, 0, 0]} scale={1.2} />

// Available moods: idle, typing, walking, watching, driving, celebrating, sleeping
// Available clothing: suit, casual, naked
```
